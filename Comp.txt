#include <vector>
#include <cmath>
#include <iostream>
#include <string>
#include <algorithm>
#include "tiffdp.hpp"

// =========================================================
// SPHERICAL HARMONICS
// =========================================================

float factorial(int n){
    float r=1.0f;
    for(int i=2;i<=n;i++) r*=i;
    return r;
}

float legendre(int l,int m,float x){
    float pmm=1.0f;

    if(m>0){
        float s=sqrtf((1-x)*(1+x));
        float f=1.0f;
        for(int i=1;i<=m;i++){
            pmm *= -f*s;
            f+=2.0f;
        }
    }

    if(l==m) return pmm;

    float p1=x*(2*m+1)*pmm;
    if(l==m+1) return p1;

    float p2=0;
    for(int ll=m+2; ll<=l; ll++){
        p2=((2*ll-1)*x*p1-(ll+m-1)*pmm)/(ll-m);
        pmm=p1;
        p1=p2;
    }
    return p2;
}

float SH(int l,int m,float th,float ph){
    float K=sqrtf((2*l+1)/(4*M_PI) *
                  factorial(l-abs(m)) /
                  factorial(l+abs(m)));

    if(m>0) return sqrtf(2)*K*cosf(m*ph)*legendre(l,m,cosf(th));
    if(m<0) return sqrtf(2)*K*sinf(-m*ph)*legendre(l,-m,cosf(th));
    return K*legendre(l,0,cosf(th));
}

// =========================================================
// SURFACE
// =========================================================

float evalSH(const std::vector<float>& c,int L,
             float th,float ph,float r0){

    float r=r0;
    int i=0;

    for(int l=0;l<=L;l++)
        for(int m=-l;m<=l;m++)
            r += c[i++] * SH(l,m,th,ph);

    return r;
}

// =========================================================
// INDEX
// =========================================================

inline int idx(int x,int y,int z,int W,int H){
    return x + W*(y + H*z);
}

// =========================================================
// BUILD SDF
// =========================================================

void buildSDF(std::vector<float>& sdf,
              int W,int H,int D,
              const std::vector<float>& c,
              int L){

    float cx=W*0.5f, cy=H*0.5f, cz=D*0.5f;

    int maxDim = std::max(W,std::max(H,D));
    float scale = 1.0f / (float)maxDim;

    float r0=0.35f;

    for(int z=0;z<D;z++)
    for(int y=0;y<H;y++)
    for(int x=0;x<W;x++){

        float X=(x-cx)*scale;
        float Y=(y-cy)*scale;
        float Z=(z-cz)*scale;

        float r=sqrtf(X*X+Y*Y+Z*Z);

        float th=acosf(Z/(r+1e-6f));
        float ph=atan2f(Y,X);

        float rs=evalSH(c,L,th,ph,r0);

        sdf[idx(x,y,z,W,H)] = r - rs;
    }
}

// =========================================================
// DENSITY
// =========================================================

inline float mu(float sdf){
    return fmaxf(0.0f, 1.0f - fabsf(sdf));
}

// =========================================================
// PROJECTION RX
// =========================================================

void projectRX(const std::vector<float>& sdf,
               int W,int H,int D,
               int w,int h,
               std::vector<float>& img){

    for(int py=0;py<h;py++)
    for(int px=0;px<w;px++){

        float u=(px/(float)w-0.5f);
        float v=(py/(float)h-0.5f);

        float att=0.0f;

        for(int z=0;z<D;z++){

            int x=(int)((u+0.5f)*W);
            int y=(int)((v+0.5f)*H);

            if(x<0||x>=W||y<0||y>=H) continue;

            float s=sdf[idx(x,y,z,W,H)];
            att += mu(s);
        }

        img[py*w+px]=expf(-0.02f*att);
    }
}

// =========================================================
// LOSS
// =========================================================

float costFunction(const std::vector<float>& sim,
                   const std::vector<float>& exp){

    float L=0.0f;

    for(size_t i=0;i<sim.size();i++){
        float a=logf(sim[i]+1e-6f);
        float b=logf(exp[i]+1e-6f);
        float d=a-b;
        L+=d*d;
    }

    return L/sim.size();
}

// =========================================================
// GRADIENT
// =========================================================

void computeGradient(std::vector<float>& grad,
                     std::vector<float>& c,
                     int L,
                     const std::vector<float>& target,
                     int W,int H,int D,
                     int w,int h){

    float eps=1e-3f;

    std::vector<float> sdf(W*H*D);
    std::vector<float> img(w*h);

    buildSDF(sdf,W,H,D,c,L);
    projectRX(sdf,W,H,D,w,h,img);

    float baseLoss=costFunction(img,target);

    grad.resize(c.size());

    for(size_t i=0;i<c.size();i++){

        float old=c[i];
        c[i]=old+eps;

        buildSDF(sdf,W,H,D,c,L);
        projectRX(sdf,W,H,D,w,h,img);

        float loss2=costFunction(img,target);

        grad[i]=(loss2-baseLoss)/eps;

        c[i]=old;
    }
}

// =========================================================
// SAVE
// =========================================================

void save(const std::vector<float>& img,
          int w,int h,
          const std::string& name){

    tiffdp<float> out(w,h,false,name);

    for(int y=0;y<h;y++)
    for(int x=0;x<w;x++)
        out.setPixel(x,y,img[y*w+x]);

    out.writeTIFF(name);
}

// =========================================================
// MAIN
// =========================================================

int main(){

    int W=64,H=64,D=64;
    int w=128,h=128;
    int L=3;

    std::vector<float> c((L+1)*(L+1),0.0f);

    // init sphère
    c[0]=0.35f;

    // image cible (exemple cercle)
    std::vector<float> target(w*h);

    for(int y=0;y<h;y++)
    for(int x=0;x<w;x++){
        float dx=(x-w/2.0f)/w;
        float dy=(y-h/2.0f)/h;
        float r=sqrtf(dx*dx+dy*dy);
        target[y*w+x]=(r<0.2f)?1.0f:0.0f;
    }

    std::vector<float> grad;

    // ================= OPTIM =================

    for(int iter=0;iter<50;iter++){

        computeGradient(grad,c,L,target,W,H,D,w,h);

        for(size_t i=0;i<c.size();i++){
            c[i] -= 0.02f * grad[i];

            // clamp
            if(c[i]>1) c[i]=1;
            if(c[i]<-1) c[i]=-1;
        }

        std::cout<<"Iter "<<iter<<"\n";
    }

    // ================= FINAL =================

    std::vector<float> sdf(W*H*D);
    std::vector<float> img(w*h);

    buildSDF(sdf,W,H,D,c,L);
    projectRX(sdf,W,H,D,w,h,img);

    save(img,w,h,"result.tif");

    std::cout<<"Done\n";
}
