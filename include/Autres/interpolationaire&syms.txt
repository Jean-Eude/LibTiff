#include "tiffdp.hpp"
#include <iostream>
#include <vector>
#include <cmath>
#include <cstdint>

int main() {

    unsigned width = 512;
    unsigned height = 512;

    // ------------------------------------------------
    // 1️⃣ grayscale sinus 8 bits
    // ------------------------------------------------
    std::vector<uint8_t> img(width * height);

    for (unsigned y = 0; y < height; y++) {
        for (unsigned x = 0; x < width; x++) {

            float v =
                0.5f * (std::sin(2 * M_PI * x / width) +
                        std::sin(2 * M_PI * y / height) + 2.0f);

            img[y * width + x] = (uint8_t)(v * 255.0f);
        }
    }

    // ------------------------------------------------
    // 2️⃣ symétrie horizontale
    // ------------------------------------------------
    std::vector<uint8_t> h(width * height);

    for (unsigned y = 0; y < height; y++) {
        for (unsigned x = 0; x < width; x++) {

            unsigned sx = width - 1 - x;
            h[y * width + x] = img[y * width + sx];
        }
    }

    // ------------------------------------------------
    // 3️⃣ symétrie verticale
    // ------------------------------------------------
    std::vector<uint8_t> v(width * height);

    for (unsigned y = 0; y < height; y++) {

        unsigned sy = height - 1 - y;

        for (unsigned x = 0; x < width; x++) {
            v[y * width + x] = h[sy * width + x];
        }
    }

    // ------------------------------------------------
    // 4️⃣ INTERPOLATION PAR AIRE (inline)
    // ------------------------------------------------
    unsigned newW = width / 2;
    unsigned newH = height / 2;

    std::vector<uint8_t> reduced(newW * newH);

    double scaleX = (double)width / newW;
    double scaleY = (double)height / newH;

    for (unsigned dy = 0; dy < newH; dy++) {
        for (unsigned dx = 0; dx < newW; dx++) {

            double x0 = dx * scaleX;
            double y0 = dy * scaleY;
            double x1 = (dx + 1) * scaleX;
            double y1 = (dy + 1) * scaleY;

            unsigned ix0 = (unsigned)std::floor(x0);
            unsigned iy0 = (unsigned)std::floor(y0);
            unsigned ix1 = (unsigned)std::ceil(x1);
            unsigned iy1 = (unsigned)std::ceil(y1);

            double sum = 0.0;
            unsigned count = 0;

            for (unsigned y = iy0; y < iy1; y++) {
                for (unsigned x = ix0; x < ix1; x++) {

                    if (x < width && y < height) {
                        sum += v[y * width + x];
                        count++;
                    }
                }
            }

            reduced[dy * newW + dx] =
                (uint8_t)(sum / (count ? count : 1));
        }
    }

    // ------------------------------------------------
    // 5️⃣ sauvegarde TIFF
    // ------------------------------------------------
    tiffdp<uint8_t> out(newW, newH, false, "final_area.tiff");
    out.setDatas(reduced);

    if (!out.writeTIFF()) {
        std::cerr << "Erreur écriture TIFF\n";
        return 1;
    }

    std::cout << "Pipeline terminé (inline area resize)\n";

    return 0;
}
