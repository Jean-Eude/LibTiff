#include "tiffdp.hpp"
#include <vector>
#include <cmath>
#include <iostream>

int main() {
    // =============================
    // 1. Création image d'entrée
    // =============================
    unsigned width = 1024;
    unsigned height = 1024;

    tiffdp<float> img(width, height, false, "sin_grayF.tiff");

    for (unsigned y = 0; y < height; y++) {
        for (unsigned x = 0; x < width; x++) {
            float value = static_cast<float>(
                (std::sin(2 * M_PI * x / width) +
                 std::sin(2 * M_PI * y / height) + 2.0) / 4.0
            );
            img.setPixel(x, y, value);
        }
    }

    if (!img.writeTIFF()) {
        std::cerr << "Erreur écriture image\n";
        return 1;
    }

    std::cout << "Image créée\n";

    // =============================
    // 2. Transformée de Radon
    // =============================
    unsigned num_angles = 180;
    unsigned num_rho = 1024;

    std::vector<std::vector<float>> radon(
        num_angles, std::vector<float>(num_rho, 0.0f)
    );

    float cx = width / 2.0f;
    float cy = height / 2.0f;
    float max_rho = std::sqrt(cx * cx + cy * cy);

    for (unsigned t = 0; t < num_angles; t++) {
        float theta = t * M_PI / num_angles;
        float cos_t = std::cos(theta);
        float sin_t = std::sin(theta);

        for (unsigned y = 0; y < height; y++) {
            for (unsigned x = 0; x < width; x++) {

                float val = img.getPixel(x, y);

                float xc = x - cx;
                float yc = y - cy;

                float rho = xc * cos_t + yc * sin_t;

                int r = static_cast<int>(
                    (rho + max_rho) * num_rho / (2 * max_rho)
                );

                if (r >= 0 && r < (int)num_rho) {
                    radon[t][r] += val;
                }
            }
        }
    }

    std::cout << "Transformée de Radon calculée\n";

    // =============================
    // 3. Normalisation (optionnelle)
    // =============================
    float max_val = 0.0f;
    for (unsigned t = 0; t < num_angles; t++)
        for (unsigned r = 0; r < num_rho; r++)
            if (radon[t][r] > max_val)
                max_val = radon[t][r];

    if (max_val > 0.0f) {
        for (unsigned t = 0; t < num_angles; t++)
            for (unsigned r = 0; r < num_rho; r++)
                radon[t][r] /= max_val;
    }

    // =============================
    // 4. Sauvegarde sinogramme
    // =============================
    tiffdp<float> sino(num_rho, num_angles, false, "radon.tiff");

    for (unsigned t = 0; t < num_angles; t++) {
        for (unsigned r = 0; r < num_rho; r++) {
            sino.setPixel(r, t, radon[t][r]);
        }
    }

    if (!sino.writeTIFF()) {
        std::cerr << "Erreur écriture sinogramme\n";
        return 1;
    }

    std::cout << "Sinogramme sauvegardé (radon.tiff)\n";

    return 0;
}
